# 操作命令
service nginx  start|stop|reload
service php7.0-fpm  start|stop|reload
service mysql  start|stop|reload

# nginx配置目录：
/etc/nginx/nginx.conf   
/etc/nginx/sites-available/default
# php.ini目录:
/etc/php/7.0/fpm/php.ini
